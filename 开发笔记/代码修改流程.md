## 详细代码修改对照

### 配置文件示例（GameBalanceConfig.gd）

```gdscript
class_name GameBalanceConfig
extends Node

static func get_max_stamina(level: int) -> int:
    if level <= 10:
        return 31 + level
    elif level <= 30:
        return int(41 + (level - 10) * 1.5)
    elif level <= 50:
        return int(71 + (level - 30) * 0.7)
    else:
        return mini(85 + (level - 50) * 0.5, 100)

static func get_stamina_cost_activity() -> Dictionary:
    return {
        "skill_1": 10,
        "skill_2_plus": 15,
        "article_1": 20,
        "article_2": 30,
        "article_3": 40,
        "article_4": 50,
        "article_5": 60,
        "maintenance_seo": 15,
        "maintenance_security": 10,
        "maintenance_design": 15,
        "maintenance_comment": 5,
        "maintenance_friend_link": 10,
    }

static func get_views_config() -> Dictionary:
    return {
        "seo_max": 100,
        "seo_base": 50,
        "seo_min": 20,
        "article_base_values": {1: 10, 2: 25, 3: 50, 4: 100, 5: 200}
    }
```

### utils.gd 修改示例

```gdscript
func get_max_stamina(level: int) -> int:
    return GameBalanceConfig.get_max_stamina(level)

func get_stamina_cost_activity() -> Dictionary:
    return GameBalanceConfig.get_stamina_cost_activity()

func calculate_day_stamina(tasks: Array) -> int:
    var total = 0
    for task in tasks:
        total += get_task_stamina_cost(task)
    return total

func can_select_task(tasks: Array, new_task: String) -> bool:
    var blogger = GDManager.get_blogger()
    var max_stamina = get_max_stamina(blogger.level)
    var total = calculate_day_stamina(tasks) + get_task_stamina_cost(new_task)
    return total <= max_stamina
```

### SEOBaseModifier.gd 修改示例

```gdscript
func apply(views: int, post: Dictionary, blogger: Dictionary) -> int:
    var seo = blogger.get("seo_value", 50)
    var article_level = post.get("article_level", 1)

    var config = GameBalanceConfig.get_views_config()
    var seo_max = config["seo_max"]
    var seo_base = config["seo_base"]
    var base_values = config["article_base_values"]

    var base_value = base_values.get(article_level, 10)
    var result = int((float(seo) / seo_max) * (seo_base + base_value))

    return max(result, 2)
```
